def binary_search(array, target, start, end):
    while start <= end:
        mid = (start + end) // 2

        if array[mid] == target:
            return mid

        elif array[mid] > target:
            end = mid - 1

        else:
            start = mid + 1
    return None

n = int(input())

parts = list(map(int, input().split()))

m = int(input())

orders = list(map(int, input().split()))

parts.sort()

for order in orders:
    if binary_search(parts, order, 0, n-1) != None:
        print("yes", end=' ')

    else:
        print("no", end=' ')
